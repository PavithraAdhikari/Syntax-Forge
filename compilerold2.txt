let editor, optimizedEditor;
function initializeEditor() {
  const ec = document.getElementById("editor");
  const ft = document.getElementById("editorFallback");
  if (typeof window.require === "function") {
    try {
      window.require.config({paths:{vs:"https://unpkg.com/monaco-editor@0.44.0/min/vs"}});
      window.require(["vs/editor/editor.main"], () => {
        editor = monaco.editor.create(ec, {
          value:"# Write your code here\n",language:"python",theme:"vs",
          automaticLayout:true,fontSize:14,fontFamily:"'JetBrains Mono',monospace",
          minimap:{enabled:false},
          scrollBeyondLastLine:false,
          padding:{top:12},
          mouseWheelZoom: true 
        });
        editor.addCommand(monaco.KeyMod.CtrlCmd|monaco.KeyCode.Equal, ()=>{
          const s=editor.getOption(monaco.editor.EditorOption.fontSize);
          editor.updateOptions({fontSize:Math.min(s+1,28)});
        });
        editor.addCommand(monaco.KeyMod.CtrlCmd|monaco.KeyCode.Minus, ()=>{
          const s=editor.getOption(monaco.editor.EditorOption.fontSize);
          editor.updateOptions({fontSize:Math.max(s-1,10)});
        });
        editor.addCommand(monaco.KeyMod.CtrlCmd|monaco.KeyCode.Digit0, ()=>{
          editor.updateOptions({fontSize:14});
        });
        editor.onDidChangeModelContent(()=>{
          vizCache={code:null,language:null,steps:null,type:null,algorithm:null};
        });
      });
      return;
    } catch(e) {}
  }

function startLiveSession() {
  robotReact("run");

  const code=getEditorCode(), language=document.getElementById("language").value;

  if(liveSocket) liveSocket.close();
  terminal&&terminal.clear();
  terminalWriteln("🔌 Connecting...");
  const wsUrl=`${location.protocol==='https:'?'wss':'ws'}://${location.host}`;
  liveSocket=new WebSocket(wsUrl);
  liveSocket.onopen=()=>{
    terminal&&terminal.clear();
    terminalWriteln("▶ Running...\r\n");
    liveSocket.send(JSON.stringify({type:"start",language,code}));
  };
  liveSocket.onmessage=(event)=>{
    const msg=JSON.parse(event.data);
    if(msg.type==="output") terminalWrite(msg.data);
    else if(msg.type==="exit"){
      if(msg.code===0){robotReact("success");fireConfetti();showXPToast();addXP(10);saveHistory(language,code);}
      else robotReact("error");
      terminalWriteln(`\r\n--- Exited (${msg.code}) ---`);
    } else if(msg.type==="error"){terminalWriteln(`\r\n❌ ${msg.err}`);robotReact("error");}
  };
  liveSocket.onclose=()=>terminalWriteln("\r\n🔌 Done.");
}

function renderVizStep(index,type){
  type=type||window._vizType||"bars";
  if(!vizSteps.length)return;
  const step=vizSteps[index];if(!step)return;
  document.getElementById("vizStepNum").innerText=`Step ${index+1} of ${vizSteps.length}`;
  document.getElementById("vizStepDesc").innerText=step.description||"";
  document.getElementById("vizLineNum").innerText=step.line||"—";
  document.getElementById("vizProgressBar").style.width=((index+1)/vizSteps.length*100)+"%";
  vizHighlightLine(step.line);
  const canvas=document.getElementById("vizCanvas");canvas.innerHTML="";
  if(type==="bars"){
    const arr=step.array||[],states=step.states||[],maxVal=Math.max(...arr,1);
    arr.forEach((val,i)=>{
      const bar=document.createElement("div");bar.className=`viz-bar ${states[i]||"default"}`;
      bar.style.height=Math.max(20,(val/maxVal)*160)+"px";
      const lbl=document.createElement("span");lbl.className="viz-bar-label";lbl.innerText=val;bar.appendChild(lbl);
      const idx=document.createElement("span");idx.className="viz-bar-idx";idx.innerText=i;bar.appendChild(idx);
      canvas.appendChild(bar);
    });
  } else {
    const arr=step.array||[],states=step.states||[];
    const wrap=document.createElement("div");wrap.className="viz-nodes";
    arr.forEach((val,i)=>{
      const node=document.createElement("div");node.className=`viz-node ${states[i]||"default"}`;
      node.innerText=val;
      if(states[i]==="active"||states[i]==="found"){
        const lbl=document.createElement("span");lbl.className="viz-node-label";
        lbl.innerText=states[i]==="found"?"✓ FOUND":"← here";node.appendChild(lbl);
      }
      wrap.appendChild(node);
      if(i<arr.length-1){const a=document.createElement("span");a.className="viz-node-arrow";a.innerText="→";wrap.appendChild(a);}
    });
    canvas.appendChild(wrap);
  }
  const vars=step.variables||{},prev=index>0?(vizSteps[index-1].variables||{}):{};
  const panel=document.getElementById("vizVarsPanel");panel.innerHTML="";
  Object.entries(vars).forEach(([k,v])=>{
    const ch=document.createElement("div");ch.className=`viz-var${prev[k]!==v?" changed":""}`;
    ch.innerHTML=`<span>${k}</span> = ${v}`;panel.appendChild(ch);
  });
}
function vizHighlightLine(lineNum){
  if(!editor||!editor.deltaDecorations||!lineNum)return;
  try{vizLineDecoration=editor.deltaDecorations(vizLineDecoration,[{
    range:new monaco.Range(lineNum,1,lineNum,1),
    options:{isWholeLine:true,className:"viz-line-highlight"}
  }]);}catch(e){}
}
function vizClearLineHighlight(){
  if(!editor||!editor.deltaDecorations)return;
  try{vizLineDecoration=editor.deltaDecorations(vizLineDecoration,[]);}catch(e){}
}
function vizNext(){if(vizCurrentStep<vizSteps.length-1){vizCurrentStep++;renderVizStep(vizCurrentStep);}else vizStopPlay();}
function vizPrev(){if(vizCurrentStep>0){vizCurrentStep--;renderVizStep(vizCurrentStep);}}
function vizFirst(){vizCurrentStep=0;renderVizStep(0);}
function vizLast(){vizCurrentStep=vizSteps.length-1;renderVizStep(vizCurrentStep);}
function vizTogglePlay(){
  const btn=document.getElementById("vizPlayBtn");
  if(vizPlayInterval){vizStopPlay();}
  else{
    btn.innerText="⏸ Pause";btn.classList.add("playing");
    vizPlayInterval=setInterval(()=>{
      if(vizCurrentStep<vizSteps.length-1){vizCurrentStep++;renderVizStep(vizCurrentStep);}
      else{vizStopPlay();robotReact("success");}
    },vizSpeed);
  }
}
function vizStopPlay(){
  if(vizPlayInterval){clearInterval(vizPlayInterval);vizPlayInterval=null;}
  const btn=document.getElementById("vizPlayBtn");
  if(btn){btn.innerText="▶ Play";btn.classList.remove("playing");}
}
function vizUpdateSpeed(val){vizSpeed=parseInt(val);if(vizPlayInterval){vizStopPlay();vizTogglePlay();}}
