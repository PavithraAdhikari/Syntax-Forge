// ===== EDITOR =====
let editor;
function initializeEditor() {
  const ec = document.getElementById("editor");
  const ft = document.getElementById("editorFallback");
  if (typeof window.require === "function") {
    try {
      window.require.config({ paths: { vs: "https://unpkg.com/monaco-editor@0.44.0/min/vs" } });
      window.require(["vs/editor/editor.main"], () => {
        editor = monaco.editor.create(ec, {
          value: "# Write your code here\n",
          language: "python",
          theme: "vs",  // light by default
          automaticLayout: true,
          fontSize: 14,
          fontFamily: "'JetBrains Mono', monospace",
          minimap: { enabled: false },
          scrollBeyondLastLine: false,
          padding: { top: 12 }
        });

        // Fix 3: Zoom with Ctrl+= and Ctrl+-
        editor.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.Equal, () => {
          const size = editor.getOption(monaco.editor.EditorOption.fontSize);
          editor.updateOptions({ fontSize: Math.min(size + 1, 28) });
        });
        editor.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.Minus, () => {
          const size = editor.getOption(monaco.editor.EditorOption.fontSize);
          editor.updateOptions({ fontSize: Math.max(size - 1, 10) });
        });
        editor.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.Digit0, () => {
          editor.updateOptions({ fontSize: 14 });
        });

        // Sync language dropdown with Monaco
        document.getElementById("language").addEventListener("change", () => {
          const lang = document.getElementById("language").value;
          const monacoLang = lang === "cpp" ? "cpp" : lang === "c" ? "c" :
            lang === "java" ? "java" : lang === "javascript" ? "javascript" :
            lang === "python" ? "python" : "sql";
          monaco.editor.setModelLanguage(editor.getModel(), monacoLang);
          document.getElementById("langLabel").innerText = lang.toUpperCase();
        });
      });
      return;
    } catch (e) {}
  }
  ec.classList.add("hidden");
  ft.classList.remove("hidden");
  editor = { getValue: () => ft.value, setValue: v => ft.value = v };
}

function getEditorCode() {
  return editor && typeof editor.getValue === "function" ? editor.getValue() : document.getElementById("editorFallback").value;
}

async function compileCode() {

let liveSocket;
function startLiveSession() {
  robotReact("run");
  const runs = parseInt(document.getElementById("statRuns").innerText || 0) + 1;
  document.getElementById("statRuns").innerText = runs;
  const code = getEditorCode();
  const language = document.getElementById("language").value;
  if (!code.trim()) { terminalWriteln("⚠️ No code to run."); return; }
  if (!terminal && !terminalFallback) initLiveTerminal();
  if (liveSocket) liveSocket.close();
  terminal && terminal.clear();
  terminalWriteln("🔌 Connecting...");
  const wsUrl = `${location.protocol === 'https:' ? 'wss' : 'ws'}://${location.host}`;
  liveSocket = new WebSocket(wsUrl);
  liveSocket.onopen = () => {
    terminal && terminal.clear();
    terminalWriteln("▶ Running...\r\n");
    liveSocket.send(JSON.stringify({ type: "start", language, code }));
  };
  liveSocket.onmessage = (event) => {
    const msg = JSON.parse(event.data);
    if (msg.type === "output") {
      terminalWrite(msg.data);
    } else if (msg.type === "exit") {
      if (msg.code === 0) {
        robotReact("success");
        fireConfetti();
        showXPToast();
        addXP(10);
        saveHistory(language, code);
      } else {
        robotReact("error");
      }
      terminalWriteln(`\r\n--- Exited (${msg.code}) ---`);
    } else if (msg.type === "error") {
      terminalWriteln(`\r\n❌ ${msg.err}`);
      robotReact("error");
    }
  };
  liveSocket.onclose = () => terminalWriteln("\r\n🔌 Done.");
}

// ===== LOAD HISTORY =====
async function loadHistory() {
  if (!window._currentUID || !window._fb) return;
  const list = document.getElementById("historyList");
  list.innerHTML = "<p style='text-align:center;color:var(--subtext);padding:1rem;font-weight:700'>Loading...</p>";
  try {
    const { collection, query, where, getDocs } = window._fb;
    const q = query(collection(window._db, "history"), where("uid", "==", window._currentUID));
    const snap = await getDocs(q);
    if (snap.empty) {
      list.innerHTML = "<p style='text-align:center;color:var(--subtext);padding:2rem;font-weight:700'>No history yet. Run some code! 🚀</p>";
      return;
    }
    const rows = snap.docs
      .sort((a,b) => new Date(b.data().timestamp) - new Date(a.data().timestamp))
      .map(d => {
        const data = d.data();
        return `<tr>
          <td>${data.programName || "—"}</td>
          <td><span class="lang-badge">${data.language.toUpperCase()}</span></td>
          <td>${new Date(data.timestamp).toLocaleString()}</td>
        </tr>`;
      }).join("");
    list.innerHTML = `<table class="hist-table">
      <tr><th>Program</th><th>Language</th><th>When</th></tr>
      ${rows}
    </table>`;
  } catch (e) { list.innerHTML = "<p style='color:red;padding:1rem'>Failed to load history.</p>"; }
}

function renderVizStep(index, type) {
  type = type || window._vizType || "bars";
  if (!vizSteps.length) return;
  const step = vizSteps[index];
  if (!step) return;

  // Update step info
  document.getElementById("vizStepNum").innerText = `Step ${index + 1} of ${vizSteps.length}`;
  document.getElementById("vizStepDesc").innerText = step.description || "";
  document.getElementById("vizLineNum").innerText = step.line || "—";

  // Update progress bar
  const pct = ((index + 1) / vizSteps.length) * 100;
  document.getElementById("vizProgressBar").style.width = pct + "%";

  // Highlight line in Monaco
  vizHighlightLine(step.line);

  // Render canvas
  const canvas = document.getElementById("vizCanvas");
  canvas.innerHTML = "";

  if (type === "bars") {
    renderBars(canvas, step);
  } else {
    renderNodes(canvas, step);
  }

  // Render variables
  renderVars(step.variables || {}, index > 0 ? (vizSteps[index-1].variables || {}) : {});
}

function renderBars(canvas, step) {
  const arr = step.array || [];
  const states = step.states || [];
  const maxVal = Math.max(...arr, 1);
  const maxHeight = 160;

  arr.forEach((val, i) => {
    const bar = document.createElement("div");
    bar.className = `viz-bar ${states[i] || "default"}`;
    const h = Math.max(20, (val / maxVal) * maxHeight);
    bar.style.height = h + "px";

    const lbl = document.createElement("span");
    lbl.className = "viz-bar-label";
    lbl.innerText = val;
    bar.appendChild(lbl);

    const idx = document.createElement("span");
    idx.className = "viz-bar-idx";
    idx.innerText = i;
    bar.appendChild(idx);

    canvas.appendChild(bar);
  });
}

function renderNodes(canvas, step) {
  const arr = step.array || [];
  const states = step.states || [];
  const wrap = document.createElement("div");
  wrap.className = "viz-nodes";

  arr.forEach((val, i) => {
    const node = document.createElement("div");
    node.className = `viz-node ${states[i] || "default"}`;
    node.innerText = val;

    // Add label if state is special
    if (states[i] === "active" || states[i] === "found") {
      const lbl = document.createElement("span");
      lbl.className = "viz-node-label";
      lbl.innerText = states[i] === "found" ? "✓ FOUND" : "← here";
      node.appendChild(lbl);
    }

    wrap.appendChild(node);

    // Arrow between nodes (except last)
    if (i < arr.length - 1) {
      const arrow = document.createElement("span");
      arrow.className = "viz-node-arrow";
      arrow.innerText = "→";
      wrap.appendChild(arrow);
    }
  });

  canvas.appendChild(wrap);
}

function renderVars(vars, prevVars) {
  const panel = document.getElementById("vizVarsPanel");
  panel.innerHTML = "";
  Object.entries(vars).forEach(([key, val]) => {
    const chip = document.createElement("div");
    const changed = prevVars[key] !== val;
    chip.className = `viz-var${changed ? " changed" : ""}`;
    chip.innerHTML = `<span>${key}</span> = ${val}`;
    panel.appendChild(chip);
  });
}

function vizHighlightLine(lineNum) {
  if (!editor || !editor.deltaDecorations || !lineNum) return;
  try {
    vizLineDecoration = editor.deltaDecorations(vizLineDecoration, [{
      range: new monaco.Range(lineNum, 1, lineNum, 1),
      options: {
        isWholeLine: true,
        className: "viz-line-highlight",
        glyphMarginClassName: "viz-line-glyph"
      }
    }]);
  } catch(e) {}
}

function vizClearLineHighlight() {
  if (!editor || !editor.deltaDecorations) return;
  try { vizLineDecoration = editor.deltaDecorations(vizLineDecoration, []); } catch(e) {}
}

function vizStopPlay() {
  if (vizPlayInterval) { clearInterval(vizPlayInterval); vizPlayInterval = null; }
  const btn = document.getElementById("vizPlayBtn");
  if (btn) { btn.innerText = "▶ Play"; btn.classList.remove("playing"); }
}

function vizUpdateSpeed(val) {
  vizSpeed = parseInt(val);
  if (vizPlayInterval) { vizStopPlay(); vizTogglePlay(); }
}

// Close visualizer on overlay click
document.getElementById("visualizerOverlay")?.addEventListener("click", (e) => {
  if (e.target === document.getElementById("visualizerOverlay")) closeVisualizer();
});

window.addEventListener("load", () => {
  initializeEditor();
  initLiveTerminal();

  // Line counter
  setInterval(() => {
    try { document.getElementById("statLines").innerText = getEditorCode().split("\n").length; } catch(e) {}
  }, 1000);

  // Timer
  let secs = 0;
  setInterval(() => {
    secs++;
    document.getElementById("statTime").innerText =
      String(Math.floor(secs/60)).padStart(2,"0") + ":" + String(secs%60).padStart(2,"0");
  }, 1000);

  // Tip rotation
  setInterval(() => {
    const b = document.getElementById("robotBubble");
    if (b && !b.classList.contains("visible")) robotSay(tips[tipIndex++ % tips.length], 5000);
  }, 45000);

  document.addEventListener("mousemove", resetIdleTimer);
  document.addEventListener("keydown", resetIdleTimer);
  resetIdleTimer();

  const isFirst = !localStorage.getItem("sf_visited");
  if (isFirst) setTimeout(() => showTutorial(), 2000);
});
